from time import sleep


def counter():
    c = 0
    while True:
        print('счетчик = ', c)
        c += 1
        yield


def printer():
    c = 0
    while True:
        if c % 3 == 0:
            print('печать ...')
        c += 1
        yield


c1 = counter()
p1 = printer()

queue = [c1, p1]

while True:
    g = queue.pop(0)
    next(g)
    queue.append(g)
    sleep(0.5)
