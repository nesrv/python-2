from math import hypot


class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def distance(self, other):
        return hypot(self.x - other.x, self.y - other.y)


p1 = Point(2, 7)
p2 = Point(12, 7)


dist = p1.distance(p2) # так ок
